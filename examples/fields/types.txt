practice test_mutating_effects (Actor) {
  num: 0..5
  var: low | med | high
  toggle: boolean
  actions: [
    {
      for: Actor
      name: "Increase"
      conditions: [
        self.num < 5
      ]
      outcomes: [
        increase self.num by 1
      ]
    }
    {
      for: Actor
      name: "Cycle"
      conditions: []
      outcomes: [
        cycle self.var by 1
      ]
    }
    {
      for: Actor
      name: "Exclaim"
      conditions: [
        self.var is #high
      ]
      outcomes: [
        broadcast "Test var is high!"
      ]
    }
    {
      for: Actor
      name: "Toggle"
      conditions: []
      outcomes: [
        set self.toggle to (not self.toggle)
        broadcast "toggle is now " + self.toggle
      ]
    }
  ]
}
